

\subsection{Виды}

\begin{frame}{Типы видео-сигнатур}

    \zbluebox{Локальные}{
        \vspace{0.5em}
        \begin{itemize}
            \item   выражают локальную информацию кадра,
            \item   напрямую работают с дескрипторами особых точек.
        \end{itemize}
        \vspace{1em}
    }

    \vspace{2em}

    \zgreenbox{Глобальные}{
        \vspace{0.5em}
        \begin{itemize}
            \item   выражают одной сигнатурой целый кадр;
            \item   выражают одной сигнатурой все видео целиком;
            \item   выражают изменении кадров во времени.
        \end{itemize}
        \vspace{1em}
    }

\end{frame}
